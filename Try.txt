local SynioxLib = loadstring(game:HttpGet("https://raw.githubusercontent.com/SynioxStudios/Syniox-FleuntGui-ByYSF/refs/heads/main/SynioxHub.txt"))()

local Window = SynioxLib:CreateWindow({
    Title = "Syniox Hub",
    SubTitle = "by YSF",
    TabWidth = 160,
    Size = UDim2.fromOffset(580, 460),
    Acrylic = true,
    Theme = "SynioxHub",
    MinimizeKey = Enum.KeyCode.LeftControl
})

local MainTab = Window:AddTab({ Title = "Farm", Icon = "rbxassetid://4483345998" })
local FarmSection = MainTab:AddSection("Settings")

_G.AutoFarm = false

FarmSection:AddToggle("FarmToggle", {
    Title = "Auto Farm", 
    Default = false, 
    Callback = function(Value)
        _G.AutoFarm = Value
        if Value then
            task.spawn(function()
                while _G.AutoFarm do
                    print("Farm aktif...")
                    task.wait(1)
                end
            end)
        end
    end
})

Window:SelectTab(1)

SynioxLib:Notify({
    Title = "Syniox Hub",

    Duration = 5
})
