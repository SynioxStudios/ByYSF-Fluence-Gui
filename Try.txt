    Color3.fromRGB(160, 0, 0), -- √úst renk
    Color3.fromRGB(40, 0, 0),  -- Orta renk
    Color3.fromRGB(0, 0, 0)    -- Alt renk
})

local Window = SynioxLib:CreateWindow({
    Title = "Syniox Hub",
    SubTitle = "v4",
    TabWidth = 160,
    Size = UDim2.fromOffset(430, 320),
    Acrylic = true,
    Theme = "SpeedHubX"
})

local AutoFarm = Window:AddTab({ Title = "Farm", Icon = "dumbbell" })

AutoFarm:AddParagraph({
    Title = "Op Farm",
    Content = "You can manage auto farm options from here."
})

_G.repToggle = false
AutoFarm:AddToggle("MainFarm", {
    Title = "üí™ Auto Farm (Equip Any tool)",
    Default = false,
    Callback = function(state)
        _G.repToggle = state
        if state then
            task.spawn(function()
                while _G.repToggle do
                    local event = game:GetService("Players").LocalPlayer:FindFirstChild("muscleEvent")
                    if event then
                        event:FireServer("rep")
                    end
                    task.wait(0.01)
                end
            end)
        end
    end
})

local function manageTool(toolName)
    task.spawn(function()
        while _G[toolName.."On"] do
            local player = game.Players.LocalPlayer
            local character = player.Character
            if character then
                local tool = player.Backpack:FindFirstChild(toolName) or character:FindFirstChild(toolName)
                
                if tool then
                    if tool.Parent ~= character then
                        tool.Parent = character
                    end
                    
                    local event = player:FindFirstChild("muscleEvent")
                    if event then
                        event:FireServer("rep")
                    else
                        tool:Activate()
                    end
                end
            end
            task.wait(0.01)
        end
    end)
end
